CREATE TABLE playlist_atual (
id_playlist int,
posicao int,
id_musica uuid,
nome text,
album text,
artista text,
PRIMARY KEY (id_playlist, posicao)
);

//Dynamic columns são colunas que não tem um padrão definido
CREATE TABLE playlist_versionada (
id_playlist int,
versao int,
modificacao text,
PRIMARY KEY(id_playlist, versao)
) WITH COMPACT STORAGE;
